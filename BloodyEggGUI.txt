-- GUI Setup
local gui = Instance.new("ScreenGui", game.CoreGui)
gui.Name = "BloodyEggGUI"

local frame = Instance.new("Frame", gui)
frame.Size = UDim2.new(0, 260, 0, 220)
frame.Position = UDim2.new(0.4, 0, 0.3, 0)
frame.BackgroundColor3 = Color3.fromRGB(25, 0, 0)

frame.Active = true

toggleBtn.TextColor3 = Color3.fromRGB(255, 0, 0)
toggleBtn.Font = Enum.Font.GothamBold
toggleBtn.TextSize = 16

local minimized = false
toggleBtn.MouseButton1Click:Connect(function()
    minimized = not minimized
    for _, v in pairs(frame:GetChildren()) do
        if v ~= toggleBtn then
            v.Visible = not minimized
        end
    end
    frame.Size = minimized and UDim2.new(0, 50, 0, 50) or UDim2.new(0, 260, 0, 220)
end)

-- Dropdown

dropdown.BackgroundColor3 = Color3.fromRGB(45, 0, 0)
dropdown.TextColor3 = Color3.fromRGB(255, 255, 255)
dropdown.Font = Enum.Font.GothamBold

local dropdownList = Instance.new("ScrollingFrame", frame)
dropdownList.Position = UDim2.new(0, 20, 0, 90)
dropdownList.Size = UDim2.new(0, 220, 0, 100)

local selected = nil
dropdown.MouseButton1Click:Connect(function()
    dropdownList.Visible = not dropdownList.Visible
end)

for eggType, pets in pairs(EggPets) do
    local btn = Instance.new("TextButton", dropdownList)
    btn.Size = UDim2.new(1, 0, 0, 20)
    btn.Text = eggType
    btn.BackgroundColor3 = Color3.fromRGB(35, 0, 0)
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.Font = Enum.Font.Gotham
    btn.TextSize = 12
    btn.MouseButton1Click:Connect(function()
        selected = eggType
        dropdown.Text = "Egg: " .. eggType
        dropdownList.Visible = false
    end)
end

-- Timer label
local timerLabel = Instance.new("TextLabel", frame)
timerLabel.Size = UDim2.new(0, 220, 0, 20)
timerLabel.Position = UDim2.new(0, 20, 0, 130)
timerLabel.BackgroundTransparency = 1
timerLabel.Text = ""
timerLabel.TextColor3 = Color3.fromRGB(255, 100, 100)

timerLabel.TextSize = 14

-- Randomize logic
local cooldown = false
local spinBtn = Instance.new("TextButton", frame)
spinBtn.Text = "üé≤ Click Randomize"
spinBtn.Size = UDim2.new(0, 220, 0, 30)
spinBtn.Position = UDim2.new(0, 20, 0, 160)
spinBtn.BackgroundColor3 = Color3.fromRGB(60, 0, 0)
spinBtn.TextColor3 = Color3.new(1, 1, 1)
spinBtn.Font = Enum.Font.GothamBold
spinBtn.TextSize = 14
spinBtn.Parent = frame

spinBtn.MouseButton1Click:Connect(function()
    if cooldown or not selected then return end
    cooldown = true
    local petList = EggPets[selected]
    local chosen = petList[math.random(1, #petList)]

    -- Show text above egg (animation)
    local billboard = Instance.new("BillboardGui")
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 2, 0)
    billboard.AlwaysOnTop = true
    billboard.Name = "EggResult"
    billboard.Adornee = character:FindFirstChild("HumanoidRootPart")
    billboard.Parent = game.CoreGui

    local label = Instance.new("TextLabel", billboard)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = "üéÅ " .. chosen
    label.TextColor3 = Color3.fromRGB(255, 50, 50)
    label.Font = Enum.Font.GothamBold
    label.TextScaled = true

    local tween = TweenService:Create(label, TweenInfo.new(1), {TextTransparency = 0.2})
    tween:Play()

    local countdown = 10
    while countdown > 0 do
        timerLabel.Text = "‚è≥ Cooldown: " .. countdown .. "s"
        countdown -= 1
        wait(1)
    end

    if billboard then
        billboard:Destroy()
    end
    timerLabel.Text = ""
    cooldown = false
end)
