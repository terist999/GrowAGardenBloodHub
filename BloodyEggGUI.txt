-- Billboard prediction
local function showBillboard(target, pet)
    local bb = Instance.new("BillboardGui", target)
    bb.Size = UDim2.new(0, 200, 0, 50)
    bb.Adornee = target
    bb.AlwaysOnTop = true
    local label = Instance.new("TextLabel", bb)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.new(1, 0.2, 0.2)
    label.Font = Enum.Font.GothamBlack
    label.TextSize = 18
    label.Text = pet
    task.delay(6, function()
        bb:Destroy()
    end)
end

-- Randomize Button logic
local cooldown = false
local spinBtn = Instance.new("TextButton", frame)
spinBtn.Text = "üé≤ Click Randomize"
spinBtn.Size = UDim2.new(0, 220, 0, 30)
spinBtn.Position = UDim2.new(0, 20, 0, 160)
spinBtn.BackgroundColor3 = Color3.fromRGB(50, 0, 0)
spinBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
spinBtn.Font = Enum.Font.GothamBold
spinBtn.TextSize = 14
spinBtn.Parent = frame

spinBtn.MouseButton1Click:Connect(function()
    if cooldown or not selected then return end
    cooldown = true

    local pets = EggPets[selected]
    local chosen = pets[math.random(1, #pets)]
    timerLabel.Text = "Result: " .. chosen

    -- Add animation (scaling effect)
    spinBtn:TweenSize(UDim2.new(0, 220, 0, 40), "Out", "Quad", 0.15, true)
    task.wait(0.15)
    spinBtn:TweenSize(UDim2.new(0, 220, 0, 30), "Out", "Quad", 0.15, true)

    -- Billboard on player (simulate egg)
    showBillboard(player.Character and player.Character:FindFirstChild("Head") or frame, chosen)

    -- Cooldown countdown
    for i = 10, 1, -1 do
        timerLabel.Text = "Cooldown: " .. i .. "s"
        task.wait(1)
    end
    timerLabel.Text = ""
    cooldown = false
end)
