local LocalPlayer = Players.LocalPlayer

-- Pets per rarity
local Pets = {
    Common = {"Bunny", "Dog", "Golden lab"},
    Uncommon = {"Black bunny", "Cat", "Chicken", "Deer"},
    Rare = {"Monkey", "Orange tabby", "Pig", "Rooster", "Spotted deer"},
    Legendary = {"Cow", "Polar bear", "Sea otter", "Silver monkey", "Turtle"},
    Mythical = {"Brown mouse", "Caterpillar", "Giant ant", "Grey mouse", "Red fox", "Snail", "Squirrel"},
    Divine = {"Dragonfly", "Queen bee", "Night owl", "Raccoon"}
}

-- Weighted rarity roll
local function randomPet()
    local rarities = {"Common", "Uncommon", "Rare", "Legendary", "Mythical", "Divine"}
    local chances = {50, 25, 15, 6, 3, 1}
    local total, roll, current = 0, 0, 0

    for _, c in ipairs(chances) do total += c end
    roll = math.random(1, total)

    for i, rarity in ipairs(rarities) do
        current += chances[i]
        if roll <= current then
            local petList = Pets[rarity]
            return petList[math.random(1, #petList)], rarity
        end
    end
end

spinBtn.Size = UDim2.new(0.8, 0, 0, 50)

spinBtn.TextColor3 = Color3.new(1, 1, 1)

-- Result Display
local function displayResult(petName, rarity)
    local eggPart = workspace:FindFirstChild("Egg") or workspace:FindFirstChildWhichIsA("Part")
    if not eggPart then return end

    local label = Instance.new("BillboardGui", eggPart)
    label.Size = UDim2.new(0, 200, 0, 50)
    label.StudsOffset = Vector3.new(0, 3, 0)
    label.AlwaysOnTop = true
    label.Adornee = eggPart

    local text = Instance.new("TextLabel", label)
    text.Size = UDim2.new(1, 0, 1, 0)
    text.Text = petName .. " (" .. rarity .. ")"
    text.BackgroundTransparency = 1
    text.TextColor3 = Color3.fromRGB(255, 255, 255)
    text.TextStrokeTransparency = 0
    text.TextScaled = true

    task.delay(5, function()
        label:Destroy()
    end)
end

-- Cooldown logic
local cooldown = false
spinBtn.MouseButton1Click:Connect(function()
    if cooldown then return end
    cooldown = true
    spinBtn.Text = "Spinning..."
    task.wait(1.5)
    local pet, rarity = randomPet()
    spinBtn.Text = "You got: " .. pet
    displayResult(pet, rarity)
    task.delay(10, function()
        spinBtn.Text = "Spin Egg"
        cooldown = false
    end)
end)

-- Minimize button toggle
local minimized = false
toggleBtn.MouseButton1Click:Connect(function()
    minimized = not minimized
    main.Size = minimized and UDim2.new(0, 50, 0, 50) or UDim2.new(0, 300, 0, 150)
    for _, child in ipairs(main:GetChildren()) do
        if child:IsA("TextButton") and child ~= toggleBtn then
            child.Visible = not minimized
        end
    end
end)

-- Keyboard toggle
UIS.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if input.KeyCode == Enum.KeyCode.RightShift then
        main.Visible = not main.Visible
    end
end)
