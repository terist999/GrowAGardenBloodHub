local function spinEgg(eggModel) local prompt = eggModel:FindFirstChildOfClass("ProximityPrompt") if prompt then -- Spin animation spawn(function() for i = 1, 30 do if eggModel:IsA("Model") and eggModel.PrimaryPart then eggModel:SetPrimaryPartCFrame(eggModel.PrimaryPart.CFrame * CFrame.Angles(0, math.rad(30), 0)) end wait(0.05) end end)

-- Random content
	local randomItems = {"Golden Plant", "Bloody Root", "Pet Worm", "Mystery Seed", "Nothing üôÅ"}
	local chosen = randomItems[math.random(1, #randomItems)]

	-- Floating text
	local billboard = eggModel:FindFirstChild("BloodTag")
	if billboard then billboard:Destroy() end

	billboard = Instance.new("BillboardGui")
	billboard.Name = "BloodTag"
	billboard.Adornee = eggModel:FindFirstChildWhichIsA("BasePart")
	billboard.Size = UDim2.new(0, 150, 0, 30)
	billboard.AlwaysOnTop = true
	billboard.Parent = eggModel

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(1, 0, 1, 0)
	label.BackgroundTransparency = 1
	label.Text = "Spun: " .. chosen
	label.TextColor3 = Color3.new(1, 0, 0)
	label.TextScaled = true
	label.Font = Enum.Font.GothamBold
	label.Parent = billboard
end

end

local btn = Instance.new("TextButton")
		btn.Size = UDim2.new(1, -10, 0, 30)
		btn.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
		btn.TextColor3 = Color3.fromRGB(255, 255, 255)
