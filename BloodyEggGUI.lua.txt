local function createFloatingText(text, position)
    local billboard = Instance.new("BillboardGui", workspace)
    billboard.Adornee = position
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.AlwaysOnTop = true

    local label = Instance.new("TextLabel", billboard)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = Color3.fromRGB(255, 0, 0)
    label.TextStrokeTransparency = 0
    label.Font = Enum.Font.GothamBold
    label.TextScaled = true

    local tween = TweenService:Create(label, TweenInfo.new(1.2), {TextTransparency = 1, TextStrokeTransparency = 1})
    tween:Play()
    game.Debris:AddItem(billboard, 1.5)
end

local function cooldownTimer(seconds)
    cooldown = true
    for i = seconds, 1, -1 do
        timerLabel.Text = "Cooldown: " .. i .. "s"
        wait(1)
    end
    timerLabel.Text = ""
    cooldown = false
end

spinBtn.MouseButton1Click:Connect(function()
    if cooldown or not selectedEgg then return end

    local pets = EggPets[selectedEgg]
    if not pets then return end

    local chosenPet = pets[math.random(1, #pets)]
    createFloatingText(chosenPet, player.Character.Head)

    spawn(function()
        cooldownTimer(10)
    end)
end)
